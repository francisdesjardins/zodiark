16  package org.zodiark.server;
17  
18  import org.atmosphere.cpr.AtmosphereResource;
19  import org.zodiark.protocol.Envelope;
20  import org.zodiark.service.Service;
21  
22  /**
23   * A distributed lightweight event bus which can encompass multiple Service instances. The event bus implements publish / subscribe and point to point messaging.
24   * Messages sent over the event bus can be of any type. For publish / subscribe, messages can be published to a {@link Service} annotated with the {@link org.zodiark.server.annotation.On#value()}
25   * using one of the {@link #ioEvent(org.zodiark.protocol.Envelope, org.atmosphere.cpr.AtmosphereResource)} and {@link #message(String, Object)}.
26   * {@link #ioEvent(org.zodiark.protocol.Envelope, org.atmosphere.cpr.AtmosphereResource)} and
27   * {@link #ioEvent(org.zodiark.protocol.Envelope, org.atmosphere.cpr.AtmosphereResource, Reply)} are used to publish I/O events to {@link Service}
28   * tha require the manipulation of {@link AtmosphereResource}. {@link #message(String, Object)} and {@link #message(String, Object, Reply)} are used to publish message to other
29   * {@link Service}
30   *
31   * {@link Service} are registered against a path, defined using the {@link org.zodiark.server.annotation.On}. There can be multiple Service registered against each path,
32   * and a particular Service can be registered against multiple paths. The event bus will route a sent message to all Service which are registered against that path.
33   * All messages sent over the bus are transient. On event of failure of all or part of the event bus messages may be lost.
34   * Applications should be coded to cope with lost messages, e.g. by resending them, and making application services idempotent.
35   * When sending a message, a {@link Reply} can be provided. If so, it will be called when the reply from the receiver has been completed.
36   * Instances of EventBus are thread-safe.
37   */
38  public interface EventBus {
39      /**
40       * Deliver an I/O event to the {@link Service} that match the {@link org.zodiark.protocol.Envelope#getMessage()}'s path value.
41       * @param e an {@link Envelope}
42       * @param r an {@link AtmosphereResource}
43       * @return this
44       */
45      EventBus ioEvent(Envelope e, AtmosphereResource r);
46      /**
47       * Deliver an I/O events to the {@link Service} that match the {@link org.zodiark.protocol.Envelope#getMessage()}'s path value.
48       * @param e an {@link Envelope}
49       * @param r an {@link AtmosphereResource}
50       * @param reply a {@link Reply} for handling the {@link Service}'s success of failure
51       * @return this
52       */
53      EventBus ioEvent(Envelope e, AtmosphereResource r, Reply reply);
54      /**
55       * Deliver a message to the {@link Service} that match the path value.
56       * @param path an {@link Envelope}
57       * @param message an {@link AtmosphereResource}
58       * @return this
59       */
60      EventBus message(String path, Object message);
61      /**
62       * Deliver a message to the {@link Service} that match the path value.
63       * @param path an {@link Envelope}
64       * @param message an {@link AtmosphereResource}
65       * @param reply a {@link Reply} for handling the {@link Service}'s success of failure
66       * @return this
67       */
68      EventBus message(String path, Object message, Reply reply);
69      /**
70       * Register a path and a {@link org.zodiark.service.Service}. This is the same as doing the {@link org.zodiark.server.annotation.On}
71       *
72       * @param path the path that will be used to map {@link Service}
73       * @param service an instance of {@link Service}
74       * @return this
75       */
76      EventBus on(String path, Service service);
77      /**
78       * Unregister {@link Service}
79       * @param path
80       * @return this
81       */
82      EventBus off(String path);
83  
84  }
