44  /**
45   * An {@link AtmosphereObjectFactory} that handles the injection of Zodiark's object. Field injection are discovered via the
46   * {@link Inject} annotation. Injection uses {@link Injectable} and {@link Extendable} implementation to discover what to inject and which
47   * default implementation class to use. Those default can be overridden by calling the {@link #extendable(Class, org.zodiark.server.ZodiarkObjectFactory.Extendable)}
48   * and {@link #injectable(Class, org.zodiark.server.ZodiarkObjectFactory.Injectable)}
49   * <p/>
50   * Injectable instances are
51   *  {@link EventBus}, {@link ObjectMapper}, {@link WowzaEndpointManager}, {@link StreamingRequest}
52   * <p/>
53   * Extendable classes are
54   *  {@link AuthConfig}, {@link PublisherConfig}, {@link SubscriberConfig}, {@link org.zodiark.service.util.RestService}
55   * <p/>
56   * Injectable and Extendable can be replaced.
57   * <p/>
58   * This class contains all default instanced used by the {@link org.zodiark.service.Service} and object used by this framework.
59   * Override this class to replace the default object injection.
60   */
61  public class ZodiarkObjectFactory implements AtmosphereObjectFactory {

67      private ScheduledExecutorService timer = Executors.newScheduledThreadPool(200);

134     @Override
135     public <T, U extends T> T newClassInstance(final AtmosphereFramework framework, Class<T> classType, Class<U> tClass) throws InstantiationException, IllegalAccessException {
136         logger.debug("About to create {}", tClass.getName());
137         if (!added.getAndSet(true) && framework !=null) {
138             framework.getAtmosphereConfig().shutdownHook(new AtmosphereConfig.ShutdownHook() {
139                 @Override
140                 public void shutdown() {
141                     timer.shutdown();
142                 }
143             });
144         }
145 
146         Class<? extends T> impl = implement(classType);
147 
148         T instance = impl != null ? impl.newInstance() : tClass.newInstance();
149 
150         Field[] fields = tClass.getDeclaredFields();
151         for (Field field : fields) {
152             if (field.isAnnotationPresent(Inject.class)) {
153                 if (field.getType().isAssignableFrom(ObjectMapper.class)) {
154                     field.set(instance, inject(ObjectMapper.class));
155                 } else if (field.getType().isAssignableFrom(EventBus.class)) {
156                     field.set(instance, inject(EventBus.class));
157                 } else if (field.getType().isAssignableFrom(RestService.class)) {
158                     field.set(instance, newClassInstance(framework, RestService.class, implement(RestService.class)));
159                 } else if (field.getType().isAssignableFrom(WowzaEndpointManager.class)) {
160                     field.set(instance, inject(WowzaEndpointManager.class));
161                 } else if (field.getType().isAssignableFrom(Context.class)) {
162                     field.set(instance, new Context() {
163 
164                         @Override
165                         public <T> T newInstance(Class<T> t) {
166                             try {
167                                 return newClassInstance(framework, t, t);
168                             } catch (Exception e) {
169                                 throw new RuntimeException(e);
170                             }
171                         }
172                     });
173                 } else if (field.getType().isAssignableFrom(AuthConfig.class)) {
174                     field.set(instance, newClassInstance(framework, AuthConfig.class, implement(AuthConfig.class)));
175                 } else if (field.getType().isAssignableFrom(PublisherConfig.class)) {
176                     field.set(instance, newClassInstance(framework, PublisherConfig.class, implement(PublisherConfig.class)));
177                 } else if (field.getType().isAssignableFrom(SubscriberConfig.class)) {
178                     field.set(instance, newClassInstance(framework, SubscriberConfig.class, implement(SubscriberConfig.class)));
179                 } else if (field.getType().isAssignableFrom(StreamingRequest.class)) {
180                     field.set(instance, inject(StreamingRequest.class));
181                 } else if (field.getType().isAssignableFrom(ScheduledExecutorService.class)) {
182                     field.set(instance, timer);
183                 }
184             }
185         }
186         return instance;
187     }

189     private <T> T inject(Class<T> clazz) {
190         Injectable<T> t = injectRepository.get(clazz);
191         if (t != null) {
192             return t.construct(clazz);
193         }
194         return null;
195     }
196 
197     private <T> Class<? extends T> implement(Class<T> clazz) {
198         Extendable<T> t = implementationRepository.get(clazz);
199         if (t != null) {
200             return t.extend(clazz);
201         }
202         return null;
203     }

211     public <T> ZodiarkObjectFactory injectable(Class<T> clazz, Injectable<T> injectable) {
