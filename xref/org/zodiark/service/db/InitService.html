16  package org.zodiark.service.db;
17  
18  import org.slf4j.Logger;
19  import org.slf4j.LoggerFactory;
20  import org.zodiark.server.Reply;
21  import org.zodiark.server.annotation.Inject;
22  import org.zodiark.server.annotation.On;
23  import org.zodiark.service.config.AuthConfig;
24  import org.zodiark.service.EndpointAdapter;
25  import org.zodiark.service.util.RESTService;
26  
27  /**
28   * Initialize a remove session in a Database/Web Service for an {@link org.zodiark.service.Endpoint}. This class
29   * use the injected {@link RESTService} to communicate with the remote endpoint.
30   */
31  @On("/db/init")
32  public class InitService extends DBServiceAdapter {
33  
34      private final Logger logger = LoggerFactory.getLogger(InitService.class);
35  
36      @Inject
37      public RESTService restService;
38  
39      @Override
40      public void reactTo(String path, Object message, Reply reply) {
41          logger.trace("Servicing {}", path);
42          if (EndpointAdapter.class.isAssignableFrom(message.getClass())) {
43              EndpointAdapter p = EndpointAdapter.class.cast(message);
44              AuthConfig config = restService.post("/init/" + p.uuid(), p.message(), AuthConfig.class);
45  
46              if (config.isAuthenticated()) {
47                  reply.ok(p);
48              } else {
49                  reply.fail(p);
50              }
51          }
52      }
53  }
