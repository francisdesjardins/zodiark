54  /**
55   * The Publisher's application logic for validating, creating and starting a {@link org.zodiark.service.session.StreamingSession}
56   */
57  @On("/publisher")
58  public class PublisherServiceImpl implements PublisherService, Session<PublisherEndpoint> {
59  
60      private final ConcurrentHashMap<String, PublisherEndpoint> endpoints = new ConcurrentHashMap<>();
61      private final Logger logger = LoggerFactory.getLogger(PublisherServiceImpl.class);
62  
63      @Inject
64      public EventBus eventBus;
65

67      public ObjectMapper mapper;

69      @Inject

72      @Override
73      public void reactTo(Envelope e, AtmosphereResource r, Reply reply) {
74          logger.trace("Handling Publisher Envelop {} to Service {}", e, r.uuid());
75          switch (e.getMessage().getPath()) {
76              case LOAD_PUBLISHER_CONFIG:
77              case CREATE_PUBLISHER_SESSION:
78                  createSession(e, r);
79                  break;
80              case VALIDATE_PUBLISHER_STREAMING_SESSION:
81                  createOrJoinStreamingSession(e);
82                  break;
83              case START_PUBLISHER_STREAMING_SESSION:
84                  startStreamingSession(e);
85                  break;
86              case FAILED_PUBLISHER_STREAMING_SESSION:
87                  errorStreamingSession(e);
88                  break;
89              case TERMINATE_STREAMING_SESSSION:
90                  String uuid = e.getMessage().getUUID();
91                  PublisherEndpoint p = endpoints.get(uuid);
92                  terminateStreamingSession(p, r);
93                  break;
94              default:
95                  throw new IllegalStateException("Invalid Message Path" + e.getMessage().getPath());
96          }
97      }

102     @Override
103     public void retrieveEndpoint(Object publisherEndpointUuid, Reply reply) {
104         if (String.class.isAssignableFrom(publisherEndpointUuid.getClass())) {
105             reply.ok(endpoints.get(publisherEndpointUuid.toString()));
106         } else {
107             reply.fail(new Exception("No publisher associated"));
108         }

121             m.setPath(ERROR_STREAMING_SESSION);

123                 m.setData(mapper.writeValueAsString(new PublisherResults("ERROR")));
124             } catch (JsonProcessingException e1) {
125                 //
126             }
127             error(e, p, m);
128         } catch (IOException e1) {
129             logger.warn("{}", e1);
130         }

144     @Override
145     public void createOrJoinStreamingSession(final Envelope e) {
146         String uuid = e.getUuid();
147         PublisherEndpoint p = retrieve(uuid);
148 
149         try {
150             p.wowzaServerUUID(mapper.readValue(e.getMessage().getData(), WowzaUUID.class).getUuid());
151         } catch (IOException e1) {
152             logger.warn("{}", e1);
153         }
154 
155         // TODO: Callback is not called at the moment as the dispatching to Wowza is asynchronous
156         eventBus.message(WOWZA_CONNECT, p, new Reply<PublisherEndpoint>() {
157             @Override
158             public void ok(PublisherEndpoint p) {
159                 // TODO: Proper Message
160                 Message m = new Message();
161                 response(e, p, m);
162             }
163 
164             @Override
165             public void fail(PublisherEndpoint p) {
166                 error(e, p, constructMessage(WOWZA_CONNECT, "error"));
167             }
168         });

171     /**
172      * {@inheritDoc}
173      */
174     @Override
175     public void startStreamingSession(final Envelope e) {
176         UUID uuid = null;
177         try {
178             uuid = mapper.readValue(e.getMessage().getData(), UUID.class);
179         } catch (IOException e1) {
180             logger.warn("{}", e1);
181         }
182 
183         PublisherEndpoint p = retrieve(uuid.getUuid());
184         eventBus.message(BEGIN_STREAMING_SESSION, p, new Reply<PublisherEndpoint>() {
185             @Override
186             public void ok(PublisherEndpoint p) {
187                 response(e, p, constructMessage(BEGIN_STREAMING_SESSION, "OK"));
188             }
189 
190             @Override
191             public void fail(PublisherEndpoint p) {
192                 error(e, p, constructMessage(BEGIN_STREAMING_SESSION, "error"));
193             }
194         }).message(BROADCASTER_CREATE, p);
195     }

197     Message constructMessage(String path, String status) {
198         Message m = new Message();
199         m.setPath(path);
200         try {
201             m.setData(mapper.writeValueAsString(new PublisherResults(status)));
202         } catch (JsonProcessingException e1) {

211     @Override
212     public void response(Envelope e, PublisherEndpoint endpoint, Message m) {
213         AtmosphereResource r = endpoint.resource();
214         Envelope newResponse = Envelope.newServerReply(e, m);
215         try {
216             r.write(mapper.writeValueAsString(newResponse));
217         } catch (JsonProcessingException e1) {
218             logger.debug("Unable to write {} {}", endpoint, m);

222     PublisherEndpoint retrieve(String uuid) {
223         PublisherEndpoint p = endpoints.get(uuid);
224         if (p == null) {
225             throw new IllegalStateException("No Publisher associated with " + uuid);
226         }
227         return p;
228     }

246      * {@inheritDoc}
247      */
248     @Override
249     public void resetEndpoint(Object publisherEndpointUuid, Reply reply) {
250         try {
251             PublisherEndpoint p = endpoints.get(publisherEndpointUuid.toString());
252             AtmosphereResource r = p.resource();
253             Message m = new Message();
254             m.setPath(PUBLISHER_ABOUT_READY);
255             m.setData(mapper.writeValueAsString(new PublisherResults("READY")));
256 
257             Envelope e = Envelope.newPublisherRequest(p.uuid(), m);
258             r.write(mapper.writeValueAsString(e));
259         } catch (JsonProcessingException e1) {
260             logger.debug("Unable to write {} {}", publisherEndpointUuid);
261             reply.fail(publisherEndpointUuid);
262         }
263     }

265     /**
266      * {@inheritDoc}
267      */
268     @Override
269     public PublisherEndpoint createSession(final Envelope e, final AtmosphereResource resource) {
270         final String uuid = e.getUuid();
271         PublisherEndpoint p = endpoints.get(uuid);
272         if (p == null) {
273             p = context.newInstance(PublisherEndpoint.class);
274             p.uuid(uuid).message(e.getMessage()).resource(resource);
275 
276             endpoints.put(uuid, p);
277             eventBus.message(DB_INIT, p, new Reply<PublisherEndpoint>() {
278                 @Override
279                 public void ok(PublisherEndpoint p) {
280                     lookupConfig(e, p);
281                 }
282 
283                 @Override
284                 public void fail(PublisherEndpoint p) {
285                     error(e, p, constructMessage(VALIDATE_PUBLISHER_STREAMING_SESSION, "error"));
286                 }
287             });
288         }
289 
290         resource.addEventListener(new OnDisconnect() {
291             @Override
292             public void onDisconnect(AtmosphereResourceEvent event) {
293                 logger.debug("Publisher {} disconnected", uuid);
294                 endpoints.remove(uuid);
295             }
296         });
297         return p;
298     }

321         eventBus.message(DB_CONFIG, p, new Reply<PublisherEndpoint>() {

325             }
326

330             }
